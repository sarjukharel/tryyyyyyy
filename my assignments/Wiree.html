import socket

target = input("Enter target IP: ")
for port in range(20, 1025):
    s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    s.settimeout(0.5)
    result = s.connect_ex((target, port))
    if result == 0:
        print(f"Port {port} is open")
    s.close()

üîç Network Analyzer

import pyshark

cap = pyshark.FileCapture('sample.pcap', 
                         display_filter='tcp.flags.syn==1 and tcp.flags.ack==0')

syn_count = {}
for pkt in cap:
    src_ip = pkt.ip.src
    syn_count[src_ip] = syn_count.get(src_ip, 0) + 1

for ip, count in syn_count.items():
    if count > 50:
        print(f"[ALERT] Possible SYN flood from {ip}: {count} SYN packets")
